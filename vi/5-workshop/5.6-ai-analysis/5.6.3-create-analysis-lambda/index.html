Tạo Lambda phân tích nội dung bằng AI

Lambda phân tích AI nhận kết quả quét từ GuardDuty thông qua Amazon SQS. Khi tài liệu có trạng thái NO_THREATS_FOUND, Lambda tải file từ Amazon S3, trích xuất nội dung và gửi đến Bedrock Mantle để đánh giá rủi ro.

Mở AWS Lambda console và chọn Create function.

Cấu hình Lambda:

  • Function name: SecureDocAIAnalysis
  • Runtime: Python 3.14
  • Architecture: x86_64
  • Execution role: Chọn role có quyền truy cập S3, DynamoDB, Secrets Manager, SQS, CloudWatch Logs và gọi Decision Engine.

Sau đó chọn Create function.

def lambda_handler(event, context):
    print("Received GuardDuty SQS event:")
    print(json.dumps(event, default=str, indent=2))

    validate_configuration()

    batch_item_failures = []

    for sqs_record in event.get("Records", []):
        message_id = sqs_record.get("messageId")

        try:
            process_sqs_record(sqs_record)

        except Exception as error:
            print(
                f"Failed message {message_id}: "
                f"{type(error).__name__}: {str(error)}"
            )

            batch_item_failures.append({
                "itemIdentifier": message_id
            })

    return {
        "batchItemFailures": batch_item_failures
    }

def process_sqs_record(sqs_record):
    raw_body = sqs_record.get("body")

    if not raw_body:
        raise ValueError(
            "SQS message không có body"
        )

    eventbridge_event = json.loads(raw_body)

    print("Decoded EventBridge event:")
    print(json.dumps(
        eventbridge_event,
        default=str,
        indent=2
    ))

    validate_eventbridge_event(
        eventbridge_event
    )

    process_guardduty_result(
        eventbridge_event
    )


def validate_eventbridge_event(event):
    if event.get("source") != "aws.guardduty":
        raise ValueError(
            "Event source không hợp lệ: "
            f"{event.get('source')}"
        )

    expected_detail_type = (
        "GuardDuty Malware Protection "
        "Object Scan Result"
    )

    if event.get(
        "detail-type"
    ) != expected_detail_type:
        raise ValueError(
            "detail-type không hợp lệ: "
            f"{event.get('detail-type')}"
        )

def process_guardduty_result(event):
    detail = event.get("detail", {})

    scan_status = detail.get(
        "scanStatus"
    )

    resource_type = detail.get(
        "resourceType"
    )

    if resource_type != "S3_OBJECT":
        raise ValueError(
            "Resource type không được hỗ trợ: "
            f"{resource_type}"
        )

    object_details = detail.get(
        "s3ObjectDetails",
        {}
    )

    result_details = detail.get(
        "scanResultDetails",
        {}
    )

    bucket_name = object_details.get(
        "bucketName"
    )

    encoded_key = object_details.get(
        "objectKey"
    )

    object_etag = object_details.get(
        "eTag"
    )

    object_version_id = object_details.get(
        "versionId"
    )

    if not bucket_name or not encoded_key:
        raise ValueError(
            "GuardDuty event thiếu bucketName "
            "hoặc objectKey"
        )

    object_key = urllib.parse.unquote_plus(
        encoded_key
    )

    malware_status = result_details.get(
        "scanResultStatus"
    )

    threats = (
        result_details.get("threats")
        or []
    )

    status_reasons = (
        result_details.get("statusReasons")
        or []
    )

    event_id = event.get("id")
    scanned_at = event.get("time")

    print(json.dumps({
        "bucket": bucket_name,
        "objectKey": object_key,
        "scanStatus": scan_status,
        "malwareStatus": malware_status,
        "threats": threats,
        "statusReasons": status_reasons
    }, default=str, indent=2))

    if bucket_name != QUARANTINE_BUCKET:
        raise ValueError(
            "Kết quả GuardDuty không thuộc "
            "Quarantine Bucket của hệ thống"
        )

    user_id, document_id, file_name = (
        parse_object_key(object_key)
    )

    updated = update_document_scan_result(
        document_id=document_id,
        scan_status=scan_status,
        malware_status=malware_status,
        threats=threats,
        status_reasons=status_reasons,
        event_id=event_id,
        scanned_at=scanned_at,
        object_etag=object_etag,
        object_version_id=object_version_id
    )

    # Nếu GuardDuty event bị gửi lặp, vẫn cho phép retry AI khi
    # malware đã sạch và aiStatus đang PENDING/NOT_STARTED/FAILED.
    if not updated:
        print(
            "GuardDuty event đã được xử lý. "
            "Kiểm tra xem AI có cần retry hay không."
        )

        retry_ai_if_needed(
            document_id=document_id,
            bucket_name=bucket_name,
            object_key=object_key,
            file_name=file_name,
            malware_status=malware_status
        )
        return

    if malware_status == "NO_THREATS_FOUND":
        process_clean_document_with_mantle(
            document_id=document_id,
            bucket_name=bucket_name,
            object_key=object_key,
            file_name=file_name
        )

    elif malware_status == "THREATS_FOUND":
        print(
            "GuardDuty phát hiện malware. "
            "Không gửi nội dung sang Bedrock Mantle."
        )

    else:
        print(
            f"GuardDuty status={malware_status}. "
            "Không gửi nội dung sang Bedrock Mantle."
        )


def parse_object_key(object_key):
    parts = object_key.split("/")

    # quarantine/{userId}/{documentId}/{fileName}
    if (
        len(parts) < 4
        or parts[0] != "quarantine"
    ):
        raise ValueError(
            "S3 key không đúng định dạng "
            "quarantine/{userId}/"
            "{documentId}/{fileName}: "
            f"{object_key}"
        )

    user_id = parts[1]
    document_id = parts[2]
    file_name = "/".join(
        parts[3:]
    )

    if not user_id:
        raise ValueError(
            "Không tìm thấy userId trong S3 key"
        )

    if not document_id:
        raise ValueError(
            "Không tìm thấy documentId trong S3 key"
        )

    if not file_name:
        raise ValueError(
            "Không tìm thấy fileName trong S3 key"
        )

    return (
        user_id,
        document_id,
        file_name
    )


def update_document_scan_result(
    document_id,
    scan_status,
    malware_status,
    threats,
    status_reasons,
    event_id,
    scanned_at,
    object_etag,
    object_version_id
):
    now = utc_now()

    document_status = map_document_status(
        malware_status
    )

    try:
        response = table.update_item(
            Key={
                "documentId": document_id
            },
            UpdateExpression="""
                SET #status = :status,
                    malwareScanStatus = :scanStatus,
                    malwareStatus = :malwareStatus,
                    malwareThreats = :threats,
                    malwareStatusReasons = :statusReasons,
                    guardDutyEventId = :eventId,
                    malwareScannedAt = :scannedAt,
                    malwareObjectETag = :objectETag,
                    malwareObjectVersionId = :versionId,
                    updatedAt = :updatedAt
            """,
            ConditionExpression="""
                attribute_exists(documentId)
                AND (
                    attribute_not_exists(guardDutyEventId)
                    OR guardDutyEventId <> :eventId
                )
            """,
            ExpressionAttributeNames={
                "#status": "status"
            },
            ExpressionAttributeValues={
                ":status": document_status,
                ":scanStatus": (
                    scan_status or "UNKNOWN"
                ),
                ":malwareStatus": (
                    malware_status or "UNKNOWN"
                ),
                ":threats": threats,
                ":statusReasons": status_reasons,
                ":eventId": (
                    event_id or "UNKNOWN"
                ),
                ":scannedAt": (
                    scanned_at or now
                ),
                ":objectETag": (
                    object_etag or "UNKNOWN"
                ),
                ":versionId": (
                    object_version_id or "NONE"
                ),
                ":updatedAt": now
            },
            ReturnValues="ALL_NEW"
        )

        print("DynamoDB scan result updated:")
        print(json.dumps(
            response.get("Attributes", {}),
            default=str,
            indent=2
        ))

        return True

    except ClientError as error:
        error_code = (
            error.response
            .get("Error", {})
            .get("Code")
        )

        if (
            error_code
            == "ConditionalCheckFailedException"
        ):
            print(
                "GuardDuty event đã được xử lý "
                "hoặc document không tồn tại."
            )

            return False

        raise


def map_document_status(malware_status):
    if malware_status == "NO_THREATS_FOUND":
        return "MALWARE_SCAN_COMPLETED"

    if malware_status == "THREATS_FOUND":
        return "MALWARE_DETECTED"

    if malware_status == "UNSUPPORTED":
        return "MANUAL_REVIEW"

    if malware_status in {
        "ACCESS_DENIED",
        "FAILED"
    }:
        return "SCAN_ERROR"

    return "SCAN_ERROR"


def retry_ai_if_needed(
    document_id,
    bucket_name,
    object_key,
    file_name,
    malware_status
):
    if malware_status != "NO_THREATS_FOUND":
        return

            "documentId": document_id
        },
        ConsistentRead=True
    )

    item = response.get(
        "Item",
        {}
    )

    ai_status = item.get(
        "aiStatus"
    )

    current_malware_status = item.get(
        "malwareStatus"
    )

    print(json.dumps({
        "message": "Checking AI retry",
        "documentId": document_id,
        "aiStatus": ai_status,
        "malwareStatus": current_malware_status
    }, default=str, indent=2))

    if (
        current_malware_status
        == "NO_THREATS_FOUND"
        and ai_status in {
            "PENDING",
            "NOT_STARTED",
            "FAILED"
        }
    ):
        process_clean_document_with_mantle(
            document_id=document_id,
            bucket_name=bucket_name,
            object_key=object_key,
            file_name=file_name
        )

def process_clean_document_with_mantle(
    document_id,
    bucket_name,
    object_key,
    file_name
):
    started = mark_ai_analyzing(
        document_id
    )

    if not started:
        print(
            "AI đã chạy, đang chạy hoặc document "
            "không đủ điều kiện để phân tích."
        )
        return

    # --------------------------------------------------------
    # STEP 1: DOWNLOAD AND EXTRACT DOCUMENT TEXT
    # --------------------------------------------------------
    try:
        s3_object = get_s3_object_bytes(
            bucket_name=bucket_name,
            object_key=object_key
        )

        extraction_result = extract_document_text(
            file_name=file_name,
            file_bytes=s3_object["body"],
            content_type=s3_object["contentType"]
        )

        text_info = normalize_document_text(
            extraction_result["text"]
        )

        text_info["detectedFileType"] = extraction_result[
            "detectedFileType"
        ]
        text_info["extractionMetadata"] = extraction_result.get(
            "metadata",
            {}
        )

        print(json.dumps({
            "message": "Document text extraction completed",
            "documentId": document_id,
            "fileName": file_name,
            "contentType": s3_object["contentType"],
            "contentLength": s3_object["contentLength"],
            "originalCharacterCount": text_info[
                "originalCharacterCount"
            ],
            "analyzedCharacterCount": text_info[
                "analyzedCharacterCount"
            ],
            "truncated": text_info["truncated"],
            "detectedFileType": text_info["detectedFileType"],
            "extractionMetadata": text_info["extractionMetadata"]
        }, default=str, indent=2))

    except DocumentExtractionError as error:
        print(
            "Document text extraction failed: "
            f"{type(error).__name__}: {str(error)}"
        )

        mark_document_extraction_error(
            document_id=document_id,
            error_message=str(error)
        )

        # Đây là lỗi dữ liệu/định dạng cố định. Không raise để tránh
        # SQS gửi lại cùng một file vô hạn.
        return

    except Exception as error:
        print(
            "Document download or extraction failed unexpectedly: "
            f"{type(error).__name__}: {str(error)}"
        )

        mark_ai_error(
            document_id=document_id,
            error_message=(
                "Document download or extraction error: "
                f"{str(error)}"
            )
        )

        # Lỗi S3 hoặc lỗi tạm thời vẫn cho SQS retry.
        raise

    # --------------------------------------------------------
    # STEP 2: INVOKE BEDROCK MANTLE AND PROCESS ITS RESPONSE
    # --------------------------------------------------------
    try:

def detect_actual_file_type(file_bytes):
    if not isinstance(file_bytes, (bytes, bytearray)):
        return "UNKNOWN"

    raw = bytes(file_bytes)

    if raw.startswith(b"%PDF-"):
        return "PDF"

    if raw.startswith(b"PK\x03\x04"):
        try:
            with zipfile.ZipFile(io.BytesIO(raw)) as archive:
                names = set(archive.namelist())

                if (
                    "[Content_Types].xml" in names
                    and "word/document.xml" in names
                ):
                    return "DOCX"

        except zipfile.BadZipFile:
            return "UNKNOWN"

        return "ZIP"

    return "TXT"


def extract_document_text(
    file_name,
    file_bytes,
    content_type
):
    extension = get_file_extension(file_name)

    normalized_content_type = (
        content_type
        or "application/octet-stream"
    ).split(";", 1)[0].strip().lower()

    actual_type = detect_actual_file_type(
        file_bytes
    )

    print(json.dumps({
        "message": "Document type validation",
        "fileName": file_name,
        "extension": extension,
        "contentType": normalized_content_type,
        "actualType": actual_type
    }, indent=2))

    if actual_type == "PDF":
        if extension != "pdf":
            raise DocumentExtractionError(
                "Nội dung thực tế là PDF nhưng tên file không có đuôi .pdf"
            )

        if normalized_content_type not in {
            "application/pdf",
            "application/octet-stream"
        }:
            raise DocumentExtractionError(
                "Content-Type không phù hợp với PDF: "
                f"{normalized_content_type}"
            )

        return extract_pdf_text(
            file_bytes
        )

    if actual_type == "DOCX":
        if extension != "docx":
            raise DocumentExtractionError(
                "Nội dung thực tế là DOCX nhưng tên file không có đuôi .docx"
            )

        if normalized_content_type not in {
            "application/vnd.openxmlformats-officedocument."
            "wordprocessingml.document",
            "application/octet-stream",
            "application/zip"
        }:
            raise DocumentExtractionError(
                "Content-Type không phù hợp với DOCX: "
                f"{normalized_content_type}"
            )

        return extract_docx_text(
            file_bytes
        )

    if actual_type == "TXT":
        if extension != "txt":
            raise DocumentExtractionError(
                "Nội dung có dạng văn bản nhưng tên file không có đuôi .txt"
            )

        if not (
            normalized_content_type.startswith("text/")
            or normalized_content_type
            in {
                "application/octet-stream",
                "binary/octet-stream"
            }
        ):
            raise DocumentExtractionError(
                "Content-Type không phù hợp với TXT: "
                f"{normalized_content_type}"
            )

        return {
            "text": extract_txt_text(file_bytes),
            "detectedFileType": "TXT",
            "metadata": {
                "encodingHandled": True
            }
        }

    raise DocumentExtractionError(
        "Không hỗ trợ hoặc không xác định được loại file. "
        "Chỉ chấp nhận TXT, PDF và DOCX."
    )


def extract_pdf_text(file_bytes):
    if PdfReader is None:
        raise DocumentExtractionError(
            "Thiếu thư viện pypdf trong Lambda package hoặc Layer"
        )

    try:
        reader = PdfReader(
            io.BytesIO(file_bytes)
        )

    except Exception as error:
        raise DocumentExtractionError(
            f"Không thể mở PDF: {str(error)}"
        ) from error

    if getattr(reader, "is_encrypted", False):
        try:
            unlock_result = reader.decrypt("")

        except Exception:
            unlock_result = 0

        if unlock_result == 0:
            raise DocumentExtractionError(
                "PDF được bảo vệ bằng mật khẩu và không thể trích xuất"
            )

    page_texts = []

    for page_index, page in enumerate(reader.pages):
        try:
            page_text = page.extract_text() or ""

        except Exception as error:
            print(
                "PDF page extraction warning: "
                f"page={page_index + 1}, error={str(error)}"
            )
            page_text = ""

        if page_text.strip():
            page_texts.append(
                f"[PAGE {page_index + 1}]\n{page_text.strip()}"
            )

    text = "\n\n".join(page_texts).strip()

    if not text:
        raise DocumentExtractionError(
            "PDF không có lớp văn bản. "
            "Có thể đây là PDF scan; hãy dùng Amazon Textract OCR."
        )

    return {
        "text": text,
        "detectedFileType": "PDF",
        "metadata": {
            "pageCount": len(reader.pages),
            "encrypted": bool(
                getattr(reader, "is_encrypted", False)
            ),
            "extractionMethod": "PYPDF"
        }
    }


def extract_docx_text(file_bytes):
    if Document is None:
        raise DocumentExtractionError(
            "Thiếu thư viện python-docx trong Lambda package hoặc Layer"
        )

    try:
        document = Document(
            io.BytesIO(file_bytes)
        )

    except Exception as error:
        raise DocumentExtractionError(
            f"Không thể mở DOCX: {str(error)}"
        ) from error

    sections = []
    paragraph_count = 0
    table_count = len(document.tables)

    for paragraph in document.paragraphs:
        value = paragraph.text.strip()

        if value:
            paragraph_count += 1
            sections.append(value)

    for table_index, table in enumerate(document.tables):
        table_rows = []

        for row in table.rows:
            cells = [
                cell.text.strip()
                for cell in row.cells
            ]

            if any(cells):
                table_rows.append(
                    " | ".join(cells)
                )

        if table_rows:
            sections.append(
                f"[TABLE {table_index + 1}]\n"
                + "\n".join(table_rows)
            )

    text = "\n\n".join(sections).strip()

    if not text:
        raise DocumentExtractionError(
            "DOCX không chứa nội dung văn bản có thể trích xuất"
        )

    return {
        "text": text,
        "detectedFileType": "DOCX",
        "metadata": {
            "paragraphCount": paragraph_count,
            "tableCount": table_count,
            "extractionMethod": "PYTHON_DOCX"
        }
    }


def extract_txt_text(file_bytes):
    """
    Decode common TXT encodings safely.
    """
    if not isinstance(file_bytes, (bytes, bytearray)):
        raise DocumentExtractionError(
            "Dữ liệu TXT không phải kiểu bytes"
        )

    if not file_bytes:
        raise DocumentExtractionError(
            "File TXT không có dữ liệu"
        )

    raw = bytes(file_bytes)

    print(json.dumps({
        "message": "Inspecting TXT encoding",
        "byteLength": len(raw),
        "firstBytesHex": raw[:20].hex(),
        "nullByteCount": raw.count(b"\x00")
    }, indent=2))

    if raw.startswith(b"\xef\xbb\xbf"):
        return raw.decode("utf-8-sig")

    if raw.startswith(b"\xff\xfe"):
        return raw.decode("utf-16-le").lstrip("\ufeff")

    if raw.startswith(b"\xfe\xff"):
        return raw.decode("utf-16-be").lstrip("\ufeff")

    null_ratio = raw.count(b"\x00") / max(len(raw), 1)

    if null_ratio >= 0.10:
        even_nulls = raw[0::2].count(0)
        odd_nulls = raw[1::2].count(0)

        candidates = (
            ("utf-16-le", "utf-16-be")
            if odd_nulls >= even_nulls
            else ("utf-16-be", "utf-16-le")
        )

        for encoding in candidates:
            try:
                decoded = raw.decode(encoding).lstrip("\ufeff")

                if decoded.count("\x00") / max(len(decoded), 1) <= 0.01:
                    print(
                        f"TXT decoded successfully with encoding={encoding}"
                    )
                    return decoded

            except UnicodeDecodeError:
                continue

    for encoding in (
        "utf-8",
        "utf-8-sig",
        "cp1252",
        "latin-1"
    ):
        try:
            decoded = raw.decode(encoding)

        except UnicodeDecodeError:
            continue

        control_count = sum(
            1
            for char in decoded
            if ord(char) < 32
            and char not in "\n\r\t\f\b"
        )

        control_ratio = control_count / max(len(decoded), 1)

        if control_ratio > 0.02:
            continue

        print(
            f"TXT decoded successfully with encoding={encoding}"
        )
        return decoded

    raise DocumentExtractionError(
        "Không thể giải mã TXT bằng encoding được hỗ trợ"
    )


def get_file_extension(file_name):
    if "." not in file_name:
        return ""

    return file_name.rsplit(
        ".",
        1
    )[-1].lower()


def normalize_document_text(text):
    if not isinstance(text, str):
        raise DocumentExtractionError(
            "Document text không hợp lệ"
        )

    normalized = (
        text
        .replace("\x00", "")
        .replace("\r\n", "\n")
        .replace("\r", "\n")
    )

    normalized = re.sub(
        r"\n{4,}",
        "\n\n\n",
        normalized
    ).strip()

    if not normalized:
        raise DocumentExtractionError(
            "Không trích xuất được nội dung văn bản"
        )

    original_count = len(normalized)

    truncated = (
        original_count
        > MAX_AI_TEXT_CHARS
    )

    normalized = normalized[
        :MAX_AI_TEXT_CHARS
    ]

    return {
        "text": normalized,
        "originalCharacterCount": original_count,
        "analyzedCharacterCount": len(
            normalized
        ),
        "truncated": truncated
    }

def get_mantle_api_key():
    global _cached_mantle_api_key

    if _cached_mantle_api_key:

    response = secrets_client.get_secret_value(
        SecretId=MANTLE_SECRET_ID
    )

    secret_string = response.get(
        "SecretString"
    )

    if not secret_string:
        raise ValueError(
            "Mantle secret không có SecretString"
        )

    try:
        secret_data = json.loads(
            secret_string
        )

        api_key = (
            secret_data.get("apiKey")
            or secret_data.get("api_key")
            or secret_data.get("token")
        )

    except json.JSONDecodeError:
        api_key = secret_string.strip()

    if not api_key:
        raise ValueError(
            "Không tìm thấy apiKey trong secret"
        )

    _cached_mantle_api_key = api_key

    return api_key

def extract_chat_completion_text(response_data):
    choices = response_data.get(
        "choices",
        []
    )

    if not choices:
        raise ValueError(
            "Mantle response không có choices"
        )

    first_choice = choices[0]

    message = first_choice.get(
        "message",
        {}
    )

    content = message.get(
        "content"
    )

    if isinstance(content, str):
        return content.strip()

    if isinstance(content, list):
        text_parts = []

        for item in content:
            if not isinstance(item, dict):
                continue

            text_value = (
                item.get("text")
                or item.get("content")
            )

            if isinstance(text_value, str):
                text_parts.append(text_value)

        return "".join(text_parts).strip()

    return ""


def get_finish_reason(response_data):
    choices = response_data.get(
        "choices",
        []
    )

    if not choices:
        return None

    return choices[0].get(
        "finish_reason"
    )

def detect_urgency_or_pressure(text_value):
    """Ignore explicitly negated urgency phrases."""
    if not text_value:
        return False

    sanitized = NEGATED_URGENCY_PATTERN.sub(
        " ",
        text_value
    )

    return bool(
        URGENCY_PATTERN.search(sanitized)
    )

    r"terraform|kms|iam role|administration|runbook|workshop|lab|"
    r"documentation|configuration|troubleshooting|decrypt password)\b",
    flags=re.IGNORECASE
)

SECRET_LIKE_PATTERN = re.compile(
    r"(?<![A-Za-z0-9])(?=[^\s]{20,80})(?=.*[A-Za-z])(?=.*\d|.*[^A-Za-z0-9\s])"
    r"[A-Za-z0-9!@#$%^&*()_+\-=\[\]{};:'\",.<>/?\\|`~]{20,80}"
)


def is_suspicious_url(url):
    try:
        parsed = urllib.parse.urlparse(url)
        host = (parsed.hostname or "").lower().strip(".")
        if not host:
            return True

        if TRUSTED_DOMAIN_PATTERN.search(host):
            return False

        if re.fullmatch(r"\d{1,3}(?:\.\d{1,3}){3}", host):
            return True

        if host.startswith("xn--"):
            return True

        combined = f"{host}{parsed.path}{parsed.query}"
        return bool(SUSPICIOUS_URL_HINT_PATTERN.search(combined))

    except Exception:
        return True


def redact_secret_like_text(value):
    if not isinstance(value, str):
        return ""

    def _redact(match):
        token = match.group(0)
        if len(token) <= 8:
            return "[REDACTED]"
        return f"{token[:4]}****{token[-4:]}"

    return SECRET_LIKE_PATTERN.sub(_redact, value)


def extract_local_security_signals(document_text):
    text_value = document_text or ""

    all_urls = list(dict.fromkeys(
        URL_PATTERN.findall(text_value)
    ))[:20]

    suspicious_urls = [
        url for url in all_urls
        if is_suspicious_url(url)
    ][:20]

    explicit_credential_request = bool(
        CREDENTIAL_REQUEST_PATTERN.search(text_value)
    )

    urgency = detect_urgency_or_pressure(
        text_value
    )

    secrecy_or_bypass = bool(
        SECRECY_PATTERN.search(text_value)
    )

    technical_context = bool(
        TECHNICAL_CONTEXT_PATTERN.search(text_value)
    )

    secret_like_value_present = bool(
        SECRET_LIKE_PATTERN.search(text_value)
    )

    strong_signal_count = sum([
        explicit_credential_request,
        urgency,
        secrecy_or_bypass,
        bool(suspicious_urls)
    ])

    return {
        "explicitCredentialRequest": explicit_credential_request,
        "urgencyOrPressure": urgency,
        "secrecyOrBypass": secrecy_or_bypass,
        "technicalContext": technical_context,
        "allLinks": all_urls,
        "suspiciousLinks": suspicious_urls,
        "suspiciousLinkPresent": bool(suspicious_urls),
        "secretLikeValuePresent": secret_like_value_present,
        "strongSignalCount": strong_signal_count
    }


def reconcile_ai_with_local_signals(ai_result, local_signals):
    """
    Keep model output separate from deterministic evidence.

    Model-only claims are not treated as independently corroborated facts.
    Ambiguous claims are downgraded to REVIEW rather than automatic REJECT.
    """
    reconciled = dict(ai_result)

    model_risk_score = reconciled["riskScore"]
    model_risk_level = reconciled["riskLevel"]
    model_action = reconciled["recommendedAction"]

    local_explicit_request = local_signals["explicitCredentialRequest"]
    local_links = local_signals["suspiciousLinks"]
    technical_context = local_signals["technicalContext"]

    reconciled["modelRiskScore"] = model_risk_score
    reconciled["modelRiskLevel"] = model_risk_level
    reconciled["modelRecommendedAction"] = model_action

    reconciled["localExplicitCredentialRequest"] = local_explicit_request
    reconciled["localUrgencyOrPressure"] = local_signals["urgencyOrPressure"]
    reconciled["localSecrecyOrBypass"] = local_signals["secrecyOrBypass"]
    reconciled["localTechnicalContext"] = technical_context
    reconciled["localSuspiciousLinks"] = local_links
    reconciled["localAllLinks"] = local_signals["allLinks"]
    reconciled["localSecretLikeValuePresent"] = local_signals[
        "secretLikeValuePresent"
    ]
    reconciled["strongSignalCount"] = local_signals["strongSignalCount"]

    reconciled["suspiciousLinks"] = list(dict.fromkeys(local_links))[:20]

    model_credential_concern = (
        reconciled["credentialTheftDetected"]
        or reconciled["explicitCredentialRequestDetected"]
    )

    has_deterministic_support = (
        local_explicit_request
        or local_signals["urgencyOrPressure"]
        or local_signals["secrecyOrBypass"]
        or bool(local_links)
    )

    model_claim_downgraded = (
        model_credential_concern
        and not has_deterministic_support
        and not reconciled["phishingDetected"]
        and not reconciled["paymentFraudDetected"]
        and not reconciled["impersonationDetected"]
    )

    reconciled["modelClaimDowngraded"] = model_claim_downgraded
    reconciled["modelCredentialConcern"] = model_credential_concern

    if model_claim_downgraded:
        reconciled["credentialTheftDetected"] = False

    system_risk_score = model_risk_score

    if (
        technical_context
        and not has_deterministic_support
        and not reconciled["phishingDetected"]
        and not reconciled["paymentFraudDetected"]
        and not reconciled["impersonationDetected"]
    ):
        system_risk_score = min(system_risk_score, 30)

    if (
        local_signals["secretLikeValuePresent"]
        and has_deterministic_support
    ):
        system_risk_score = max(system_risk_score, 35)

    if model_claim_downgraded:
        system_risk_score = min(max(system_risk_score, 31), 50)

    reconciled["systemRiskScore"] = system_risk_score
    reconciled["systemRiskLevel"] = score_to_risk_level(system_risk_score)

    # Backward-compatible fields now represent the system assessment.
    reconciled["riskScore"] = system_risk_score
    reconciled["riskLevel"] = reconciled["systemRiskLevel"]

    reconciled["evidence"] = [
        redact_secret_like_text(item)
        for item in reconciled.get("evidence", [])
    ]
    reconciled["summary"] = redact_secret_like_text(
        reconciled.get("summary", "")
    )

    return reconciled


def calculate_ai_decision(ai_result):
    """
    ALLOW -> automatic SAFE
    REJECT -> automatic REJECTED
    REVIEW -> admin handles only genuinely difficult cases
    """
    score = ai_result["systemRiskScore"]
    model_score = ai_result["modelRiskScore"]
    confidence = ai_result["confidence"]
    model_action = ai_result["modelRecommendedAction"]

    local_explicit_request = ai_result["localExplicitCredentialRequest"]
    urgency = ai_result["localUrgencyOrPressure"]
    secrecy = ai_result["localSecrecyOrBypass"]
    suspicious_links = bool(ai_result["localSuspiciousLinks"])

    phishing = ai_result["phishingDetected"]
    social_engineering = ai_result["socialEngineeringDetected"]
    credential_concern = ai_result["modelCredentialConcern"]
    deceptive_intent = ai_result["deceptiveIntentDetected"]
    payment_fraud = ai_result["paymentFraudDetected"]
    impersonation = ai_result["impersonationDetected"]
    prompt_injection = ai_result["promptInjectionDetected"]

    deterministic_support_count = sum([
        local_explicit_request,
        urgency,
        secrecy,
        suspicious_links
    ])

    strong_model_signal_count = sum([
        phishing,
        social_engineering,
        credential_concern,
        deceptive_intent,
        payment_fraud,
        impersonation,
        prompt_injection
    ])

    model_rule_disagreement = (
        model_action == "ALLOW"
        and deterministic_support_count > 0
    ) or (
        model_action == "REJECT"
        and deterministic_support_count == 0
    )

    ai_result["deterministicSupportCount"] = deterministic_support_count
    ai_result["strongModelSignalCount"] = strong_model_signal_count
    ai_result["modelRuleDisagreement"] = model_rule_disagreement

    if (
        score >= AUTO_REJECT_MIN_SCORE
        and confidence >= MIN_AUTO_DECISION_CONFIDENCE
        and deterministic_support_count >= 2
        and strong_model_signal_count >= 1
    ):
        ai_result["reviewReasonCode"] = "NONE"
        ai_result["systemDecisionReason"] = (
            "Automatic rejection: critical risk was corroborated by "
            "multiple deterministic indicators and a high-confidence "
            "malicious model finding."
        )
        return "REJECT"

    if (
        local_explicit_request
        and confidence >= 0.85
        and (
            suspicious_links
            or secrecy
            or impersonation
            or phishing
        )
    ):
        ai_result["reviewReasonCode"] = "NONE"
        ai_result["systemDecisionReason"] = (
            "Automatic rejection: an explicit credential request was "
            "corroborated by phishing, impersonation, secrecy, or a "
            "suspicious link."
        )
        return "REJECT"

    if (
        payment_fraud
        and confidence >= 0.90
        and deterministic_support_count >= 2
    ):
        ai_result["reviewReasonCode"] = "NONE"
        ai_result["systemDecisionReason"] = (
            "Automatic rejection: high-confidence payment fraud was "
            "corroborated by multiple deterministic indicators."
        )
        return "REJECT"

    if (
        score <= AUTO_ALLOW_MAX_SCORE
        and model_score <= AUTO_ALLOW_MAX_SCORE
        and model_action == "ALLOW"
        and confidence >= MIN_AUTO_DECISION_CONFIDENCE
        and strong_model_signal_count == 0
        and deterministic_support_count == 0
        and not model_rule_disagreement
    ):
        ai_result["reviewReasonCode"] = "NONE"
        ai_result["systemDecisionReason"] = (
            "Automatic allow: GuardDuty found no malware, the model "
            "recommended ALLOW with sufficient confidence, and no "
            "deterministic malicious indicator was detected."
        )
        return "ALLOW"

    review_reasons = []

    if confidence < MIN_AUTO_DECISION_CONFIDENCE:
        review_reasons.append("LOW_AI_CONFIDENCE")

    if model_rule_disagreement:
        review_reasons.append("MODEL_RULE_DISAGREEMENT")

    if ADMIN_REVIEW_MIN_SCORE <= score < AUTO_REJECT_MIN_SCORE:
        review_reasons.append("AMBIGUOUS_RISK_SCORE")

    if model_action == "REVIEW":
        review_reasons.append("MODEL_RECOMMENDED_REVIEW")

    if deterministic_support_count == 1 and strong_model_signal_count >= 1:
        review_reasons.append("SINGLE_CORROBORATING_INDICATOR")

    if ai_result.get("modelClaimDowngraded"):
        review_reasons.append("UNSUPPORTED_MODEL_CLAIM")

    if strong_model_signal_count > 0 and deterministic_support_count == 0:
        review_reasons.append("MODEL_ONLY_SECURITY_FINDING")

    if deterministic_support_count > 0 and strong_model_signal_count == 0:
        review_reasons.append("RULE_ONLY_SECURITY_FINDING")

    if not review_reasons:
        review_reasons.append("UNCLASSIFIED_AMBIGUITY")

    ai_result["reviewReasonCode"] = "|".join(
        list(dict.fromkeys(review_reasons))
    )
    ai_result["systemDecisionReason"] = (
        "Administrator review required because the evidence is ambiguous, "
        "insufficiently corroborated, low-confidence, or inconsistent "
        "between the model and deterministic rules."
    )
    return "REVIEW"

def parse_ai_json(raw_text):
    cleaned = raw_text.strip()

    cleaned = re.sub(
        r"^```(?:json)?\s*",
        "",
        cleaned,
        flags=re.IGNORECASE
    )

    cleaned = re.sub(
        r"\s*```$",
        "",
        cleaned
    )

    try:
        result = json.loads(cleaned)

    except json.JSONDecodeError:
        first_brace = cleaned.find("{")
        last_brace = cleaned.rfind("}")

        if (
            first_brace == -1
            or last_brace <= first_brace
        ):
            raise ValueError(
                "Mantle response không chứa JSON object"
            )

        json_text = cleaned[
            first_brace:last_brace + 1
        ]

        try:
            result = json.loads(
                json_text
            )

        except json.JSONDecodeError as error:
            raise ValueError(
                "Mantle trả JSON không hợp lệ"
            ) from error

    if not isinstance(result, dict):
        raise ValueError(
            "Kết quả AI phải là JSON object"
        )

    return result

):

def normalize_bool(value):
    if isinstance(value, bool):
        return value

    if isinstance(value, str):
        return value.strip().lower() in {
            "true",
            "yes",
            "1"
        }

    return bool(value)


def normalize_string(
    value,
    maximum_length
):
    if not isinstance(value, str):
        return ""

    return value.strip()[
        :maximum_length
    ]


def normalize_string_list(
    value,
    maximum_items,
    maximum_item_length
):
    if not isinstance(value, list):
        return []

    output = []

    for item in value[
        :maximum_items
    ]:
        if not isinstance(item, str):
            continue

        cleaned = item.strip()[
            :maximum_item_length
        ]

        if cleaned:
            output.append(cleaned)

    return output
